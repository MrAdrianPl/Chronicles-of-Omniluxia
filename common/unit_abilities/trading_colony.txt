trading_colony = {

	duration = 0
	enable = yes
	toggle = no
	army_only = yes

	map = yes
	soundeffect ="event:/SFX/Animations/Units/Generic/sfx_anim_unit_build_colony"
	
	allow = {
		unit_owner = {
			can_pay_price = price_trade_colony
			manpower >= 3
			OR = {
				has_country_modifier = trade_2
				has_country_modifier = trade_3
				has_country_modifier = trade_4
			}
        }		
		in_combat = no
		in_siege = no
		is_exiled = no
		custom_tooltip = {
			text = "owns_no_neighboring_province"
			unit_location = { 
				is_inhabitable = yes
				has_owner = no
				NOT = {
					any_neighbor_province = {
						owner = root.unit_owner
					}
				}
			}
		}		
        hidden:unit_location = { NOT = { is_sea = yes } }
		has_commander = yes
		is_army = yes
		is_moving = no
		custom_tooltip = {
			text = "num_of_loyal_cohorts_tooltip"
			num_of_non_loyal_cohorts_svalue >= 1
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	start_effect = {
		if = {
			limit = {
				unit_location = { 
		            is_inhabitable = yes
		            has_owner = no
		        }
			}
			unit_location = {

				set_owned_by = root.unit_owner

	            root.unit_owner = {
					pay_price = price_trade_colony
					add_manpower = -3
				}
			}
		}
	}
}
